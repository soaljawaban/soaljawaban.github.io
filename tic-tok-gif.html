12. Apa kelebihan dan kekurangan gif


GIF: Graphics Interchange Format

PROS
- 256 jumlah warna
- menggunakan kompresi lossless
- Cocok untuk logo, bidang warna datar dengan daerah yang terdefinisi dengan baik
- Cocok untuk ikon dengan beberapa warna
- mendukung transparansi
- Cocok untuk animasi kecil

CONS
- format tertua untuk web - 1989
- dalam banyak kasus memiliki ukuran file yang lebih besar dari PNG3. GIF (Graphics Interchange Format)
Type file GIF memungkinkan penambahan warna transparan dan dapat digunakan untuk membuat animasi sederhana, tetapi sekarang ini standar GIF hanya maksimal 256 warna saja.
Kelebihan :

Mendukung animasi gambar
Bersifat tidak pecah-pecah
Menggunakan kompresi tidak menghilangkan data


Kekurangan :

Tidak cocok digunakan dalam fotografi modern
Penyimpanan data gambar terbatas
Gambar GIF berukuran lebih kecil dari pada JPG

17. apa yang di maksud extension*.gif

Semoga Membantu!

Ekstensi file GIF (Graphics Interchange Format) adalah format gambar bitmap yang dirilis oleh CompuServe pada tahun 1987.

Format gambar ini mendukung sampai dengan 8 bit per piksel, memungkinkan sebuah gambar untuk merefrensikan palet warnanya sendiri sampai dengan 256 warna berbeda.

Fitur GIF yang paling dikenal adalah ekstensi GIF memungkinkan animasi atau gambar bergerak.

#include <iostream>

using namespace std;

char papan[3][3] = {{'1', '2', '3'}, {'4', '5', '6'}, {'7', '8', '9'}};

char giliran = 'X';

bool selesai = false;

void TampilkanPapan() {

system("cls"); // Menghapus layar untuk tampilan yang lebih baik

cout << "Tic Tac Toe\n\n";

for (int i = 0; i < 3; i++) {

for (int j = 0; j < 3; j++) {

cout << papan[i][j];

if (j < 2) {

cout << " | ";

}

}

cout << endl;

if (i < 2) {

cout << "---------\n";

}

}

cout << endl;

}

void GantiGiliran() {

if (giliran == 'X') {

giliran = 'O';

} else {

giliran = 'X';

}

}

bool LakukanLangkah(int langkah) {

for (int i = 0; i < 3; i++) {

for (int j = 0; j < 3; j++) {

if (papan[i][j] == '0' + langkah) {

if (papan[i][j] == 'X' || papan[i][j] == 'O') {

return false; // Langkah sudah diambil

}

papan[i][j] = giliran;

return true;

}

}

}

return false; // Langkah tidak valid

}

bool CekKemenangan() {

for (int i = 0; i < 3; i++) {

if (papan[i][0] == papan[i][1] && papan[i][1] == papan[i][2]) {

return true; // Pemenang horizontal

}

}

for (int i = 0; i < 3; i++) {

if (papan[0][i] == papan[1][i] && papan[1][i] == papan[2][i]) {

return true; // Pemenang vertikal

}

}

if (papan[0][0] == papan[1][1] && papan[1][1] == papan[2][2]) {

return true; // Pemenang diagonal (kiri atas - kanan bawah)

}

if (papan[0][2] == papan[1][1] && papan[1][1] == papan[2][0]) {

return true; // Pemenang diagonal (kanan atas - kiri bawah)

}

return false;

}

int main() {

int langkah;

TampilkanPapan();

while (!selesai) {

cout << "Giliran pemain " << giliran << ". Masukkan langkah (1-9): ";

cin >> langkah;

if (langkah < 1 || langkah > 9) {

cout << "Langkah tidak valid. Coba lagi.\n";

continue;

}

if (!LakukanLangkah(langkah)) {

cout << "Langkah sudah diambil. Coba lagi.\n";

continue;

}

TampilkanPapan();

if (CekKemenangan()) {

cout << "Pemain " << giliran << " memenangkan permainan!\n";

selesai = true;

} else if (papan[0][0] != '1' && papan[0][1] != '2' && papan[0][2] != '3' &&

papan[1][0] != '4' && papan[1][1] != '5' && papan[1][2] != '6' &&

papan[2][0] != '7' && papan[2][1] != '8' && papan[2][2] != '9') {

cout << "Permainan berakhir dengan hasil seri!\n";

selesai = true;

} else {

GantiGiliran();

}

}

return 0;

}
